$.mergeData(target,src)

描述：

合并src的缓存数据到target的缓存数据上。

参数：

target
必需。目标对象。
src
可选。被合并数据的对象。
例子 
                $.require("data,spec",function(){
                    var a = {};
                    //写入数揣
                    $.data(a,"name","司徒正美");
                    //读取数据
                    $.log($.dump(a),true);//{ uniqueNumber:2, "@data_2":{data:{name:"司徒正美"}} }
                    var b = {};
                    $.data(b,"sex","man");
                    //合并数据
                    $.mergeData(b,a);
                    $.log($.dump(b),true);//{ uniqueNumber:3, "@data_3":{data:{name:"司徒正美",sex:"man"}} }
                });
